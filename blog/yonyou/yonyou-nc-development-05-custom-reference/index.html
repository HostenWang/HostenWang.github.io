3 代码实现

基于重载的实现(不建议)

package nc.ui.pm.ref;
 
import nc.ui.bd.ref.AbstractRefModel;
import nc.vo.pubapp.pattern.pub.SqlBuilder;
 
/**
 * 合同档案参照类 2021-08-12 11:15:04
 *
 * @author Hosten
 */
public class ContractRef extends AbstractRefModel {
    /**
     * 参照列名称
     *
     */
    @Override
    public String getRefTitle() {
        return "合同类型参照";
    }
 
    /**
     * 可显示字段编码
     */
    private static final String[] FIELD_CODE = { "contractnum", "contractname",
            "contracttype", "pk_pm_h" };
 
    /**
     * 可显示字段
     */
    @Override
    public String[] getFieldCode() {
        return ContractRef.FIELD_CODE;
    }
 
    /**
     * 参照列名称
     *
     */
    private static final String[] FIELDNAME = { "合同号", "合同方", "合同类型", "主键" };
     
    /**
     * 显示列名
     */
    @Override
    public int getDefaultFieldCount() {
        return ContractRef.FIELDNAME.length;
    }
 
    @Override
    public String[] getFieldName() {
        return ContractRef.FIELDNAME;
    }
 
    private static final String table = "hzctr_pm_h";
 
    @Override
    public String getTableName() {
        return ContractRef.table;
    }
 
    @Override
    protected String getEnvWherePart() {
        SqlBuilder wherePart = new SqlBuilder();
        wherePart.append(" nvl(dr,0)=0 and 1=1  ");
        return wherePart.toString();
    }
 
    /**
     * 参照列主键
     */
    public static final String PKCODE = "pk_pm_h";
 
    @Override
    public String getPkFieldCode() {
        return ContractRef.PKCODE;
    }
 
    /**
     * 隐藏列
     */
    private static final String[] HIDDENFIELDCODE = { "pk_pm_h" };
 
    @Override
    public String[] getHiddenFieldCode() {
        return ContractRef.HIDDENFIELDCODE;
    }
}

基于构造函数的实现(推荐)

package nc.ui.gl.ref;

import nc.ui.bd.ref.AbstractRefModel;

public class TestJiluRefModel extends AbstractRefModel {

	public TestJiluRefModel(String refNodeName) {
		setRefNodeName(refNodeName);
	}

	public void setRefNodeName(String refNodeName) {
		m_strRefNodeName = refNodeName;
		// 以部门表为例，只是举例，非真实需求
		setFieldCode(new String[] { "pk_deptdoc", "deptname", "deptcode" });
		setFieldName(new String[] { "部门名称", "部门编码", "树型结构码" });
		setHiddenFieldCode(new String[] { "bd_deptdoc.remcode", "pk_fathedept",
				"showorder" });
		setPkFieldCode("pk_deptdoc");
		setRefCodeField("pk_deptdoc");
		setRefNameField("deptname");
		setTableName("bd_deptdoc");
		setWherePart("( pk_corp='" + getPk_corp() + "') ");
		setOrderPart(" showorder,deptcode ");
		setRefTitle("部门档案");
		resetFieldName();
	}
}

单据模板设置

在单据模板初始化那里找到相应的单据模板； 选择字段后，修改字段类型为参照，点击类型设置在“参照选择”可以找到“自定义参照名称”， 不需要 在 是否自定义参照 打勾。

4 使用

获取参照的值

UIRefPane measdocRef = (UIRefPane) getBillCardPanel().getHeadItem(MaterialVO.PK_MEASDOC).getComponent();
measdocRef.getRefPK(); //主键 
measdocRef.getRefName(); //名称
measdocRef.getRefCode(); //编码

参照的过滤条件

//可以添加在编辑前事件中
UIRefPane refPane=(UIRefPane)billForm.getBillCardPanel().getHeadItem("def2").getComponent();
AbstractRefModel refModel=(AbstractRefModel) refPane.getRefModel();
String wherePart=" and def5='1001H2100000004MJ7SV' ";//Sql以and开头
refModel.addWherePart(wherePart, false);//第二个参数为是否刷新该数据
refPane.setMultiCorpRef(false);//设置不能跨组织

表头参照多选取值

((UIRefPane)billCardPanel.getHeadItem("hzdy41").getComponent()).getRefModel().getPkValues();

动态指定参照类

((UIRefPane)(getBillCardPanelWrapper().getBillCardPanel().getHeadItem("字段").getComponent())).setRefModel(参照类.class);

设置参照是否支持多选

((UIRefPane)editor.getBillCardPanel().getHeadItem("vpsn").getComponent()).setMultiSelectedEnabled(true);

判断参照类型

Object obj =((UIRefPane)getBillCardPanel().getHeadItem("").getComponent()).getRefModel();
