使用 WFS Transaction 修改要素

<body>
    <button id="save">保存修改</button>
    <div id="map"></div>

    <script>
      	// 用于保存被修改的要素
        let modifiedFeatures = null;    
        let save = document.getElementById('save');

        let vectorLayer = new ol.layer.Vector({
            source: new ol.source.Vector({
              	// // 用于解析从服务器端发过来的GeoJSON数据
                format: new ol.format.GeoJSON({    
                    geometryName: 'geom'           // 注意：这里图形信息字段一定要和后端服务器一致
                }),            
                // 设置KVP格式的请求URL
                url: 'http://localhost:8080/geoserver/wfs?SERVICE=WFS&VERSION=1.1.0&REQUEST=GetFeature' + 
                    '&TYPENAME=demo:world&OUTPUTFORMAT=application/json'
            })
        });

        let map = new ol.Map({
            target: 'map',
            layers: [
                new ol.layer.Tile({
                    source: new ol.source.Stamen({
                        layer: 'world'
                    })
                }),
                vectorLayer
            ],
            view: new ol.View({
                projection: 'EPSG:3857',
                center: [114, 80],
                zoom: 3
            })
        });

        // 选择要素控件
        let select = new ol.interaction.Select({
            layers: [
                vectorLayer
            ]
        });
        map.addInteraction(select);

        // 修改要素控件
        let modify = new ol.interaction.Modify({
            features: select.getFeatures()
        });
        // 保存修改完成的要素
        modify.on('modifyend', (event) => {
            modifiedFeatures = event.features;
        });
        map.addInteraction(modify);

        save.addEventListener('click', (event) => {
            let features = [];
            modifiedFeatures.forEach(function(item, index, array){
                features.push(item);
            });
            updateWFS(features);
        });

        function updateWFS(features){
            let WFS = new ol.format.WFS();
            // 生成XML格式的WFS请求信息
            let transact_xml = WFS.writeTransaction(
                null, features, null, 
                {
                    featureNS: 'http://localhost:8084/geoserver/demo',
                    featureType: 'world',
                    srsName: 'EPSG:3857'
                }
            )
            // 将XML格式请求信息序列化为字符串格式
            transact_str = (new XMLSerializer()).serializeToString(transact_xml); 
            // 使用AJAX将请求发送到后端
            let xhr = new XMLHttpRequest();
            xhr.addEventListener('readystatechange', (event) => {
                if(xhr.readyState === 4){
                    if((xhr.status === 200) || xhr.status === 304){
                        let transactRes = WFS.readTransactionResponse(xhr.responseText);
                        let str = transactRes.transactionSummary.totalInserted + 
                                   " totalInserted!, insertIds: " + transactRes.insertIds + "\n";
                        str += transactRes.transactionSummary.totalUpdated + " totalUpdated!\n";
                        str += transactRes.transactionSummary.totalDeleted + " totalDeleted!";
                        alert(str);
                    }else{
                        alert("修改失败!");
                    }
                }
            })
            xhr.open('POST', 'http://localhost:8080/geoserver/wfs');
            xhr.setRequestHeader('Content-Type', 'text/xml');
            xhr.send(transact_str);
        }
    </script>
</body>

使用 WFS Transaction 添加要素

<body>
    <button id="add">添加</button>
    <div id="map"></div>

    <script>
        let drawedFeatures = [];
        let addButton = document.getElementById('add');

        let vectorLayer = new ol.layer.Vector({
            source: new ol.source.Vector({
                format: new ol.format.GeoJSON({
                    geometryName: 'geom'
                }),            
                url: 'http://localhost:8080/geoserver/wfs?SERVICE=WFS&VERSION=1.1.0&REQUEST=GetFeature' + 
                    '&TYPENAME=demo:world&OUTPUTFORMAT=application/json'
            })
        });

        let map = new ol.Map({
            target: 'map',
            layers: [
                new ol.layer.Tile({
                    source: new ol.source.Stamen({
                        layer: 'world'
                    })
                }),
                vectorLayer
            ],
            view: new ol.View({
                projection: 'EPSG:3857',
                center: [114, 80],
                zoom: 3
            })
        });

        // 绘制控件
        let draw = new ol.interaction.Draw({
            source: vectorLayer.getSource(),
            type: 'MultiPolygon',
            geometryName: 'geom'
        });
        draw.on('drawend', (event) => {
            let feature = event.feature;
            feature.set('name', '格林兰岛');
            feature.set('popname', '');
            feature.set('code', '');
            feature.set('type', 0);
            feature.set('display', 0);
            feature.set('extention', 0);
            feature.set('update', '');
            feature.set('x', 90);
            feature.set('y', 80);
            drawedFeatures.push(event.feature);
        });
        map.addInteraction(draw);

        addButton.addEventListener('click', (event) => {
            // 将绘制的要素添加到后台
            addFeatureWFS(drawedFeatures);
        });

        function addFeatureWFS(features){
            let WFS = new ol.format.WFS();
            // 生成XML格式的WFS请求信息
            let transact_xml = WFS.writeTransaction(
                features, null, null, 
                {
                    featureNS: 'http://localhost:8080/geoserver/demo',
                    featurePrefix: 'demo',
                    featureType: 'world',
                    srsName: 'EPSG:3857'
                }
            );
            // 将XML格式请求信息序列化为字符串格式
            transact_str = (new XMLSerializer()).serializeToString(transact_xml); 
            // 使用AJAX将请求发送到后端
            let xhr = new XMLHttpRequest();
            xhr.addEventListener('readystatechange', (event) => {
                if(xhr.readyState === 4){
                    if((xhr.status === 200) || xhr.status === 304){
                        let transactRes = WFS.readTransactionResponse(xhr.responseText);
                        let str = transactRes.transactionSummary.totalInserted + 
                                 " totalInserted!, insertIds: " + transactRes.insertIds + "\n";
                        str += transactRes.transactionSummary.totalUpdated + " totalUpdated!\n";
                        str += transactRes.transactionSummary.totalDeleted + " totalDeleted!";
                        alert(str);
                        drawedFeatures = [];
                    }else{
                        alert("修改失败!");
                    }
                }
            })
            xhr.open('POST', 'http://localhost:8080/geoserver/wfs');
            xhr.setRequestHeader('Content-Type', 'text/xml');
            xhr.send(transact_str);
        }
    </script>
</body>

使用 WFS Transaction 删除要素

<body>
    <button id="delete">确认删除</button>
    <div id="map"></div>

    <script>
        let deleteBtn = document.getElementById('delete');

        let vectorLayer = new ol.layer.Vector({
            source: new ol.source.Vector({
                format: new ol.format.GeoJSON({
                    geometryName: 'geom'
                }),            
                // 设置KVP格式的请求URL
                url: 'http://localhost:8080/geoserver/wfs?SERVICE=WFS&VERSION=1.1.0&REQUEST=GetFeature' + 
                    '&TYPENAME=demo:world&OUTPUTFORMAT=application/json'
            })
        });

        let map = new ol.Map({
            target: 'map',
            layers: [
                new ol.layer.Tile({
                    source: new ol.source.Stamen({
                        layer: 'watercolor'
                    })
                }),
                vectorLayer
            ],
            view: new ol.View({
                projection: 'EPSG:3857',
                center: [90, 80],
                zoom: 3
            })
        });

        // 选择要素控件
        let select = new ol.interaction.Select({
            layers: [
                vectorLayer
            ]
        });
        map.addInteraction(select);

        deleteBtn.addEventListener('click', (event) => {
            let features = [];
            select.getFeatures().forEach((item, index, array) => {
                features.push(item);
                vectorLayer.getSource().removeFeature(item);
            });
            // 向后端发送请求，删除要素
            deleteWFS(features);
        });

        function deleteWFS(features){
            let WFS = new ol.format.WFS();
            // 生成XML格式的WFS请求信息
            let transact_xml = WFS.writeTransaction(
                null, null, features, 
                {
                    featureNS: 'http://localhost:8080/geoserver/test',
                    featurePrefix: 'demo',
                    featureType: 'world',
                    srsName: 'EPSG:3857'
                }
            )
            // 将XML格式请求信息序列化为字符串格式
            transact_str = (new XMLSerializer()).serializeToString(transact_xml); 
            // 使用AJAX将请求发送到后端
            let xhr = new XMLHttpRequest();
            xhr.addEventListener('readystatechange', (event) => {
                if(xhr.readyState === 4){
                    if((xhr.status === 200) || xhr.status === 304){
                        let transactRes = WFS.readTransactionResponse(xhr.responseText);
                        let str = transactRes.transactionSummary.totalInserted + 
                          " totalInserted!, insertIds: " + transactRes.insertIds + "\n";
                        str += transactRes.transactionSummary.totalUpdated + " totalUpdated!\n";
                        str += transactRes.transactionSummary.totalDeleted + " totalDeleted!";
                        alert(str);
                        map.render();
                    }else{
                        alert("修改失败!");
                    }
                }
            })
            xhr.open('POST', 'http://localhost:8080/geoserver/wfs');
            xhr.setRequestHeader('Content-Type', 'text/xml');
            xhr.send(transact_str);
        }
    </script>
</body>
