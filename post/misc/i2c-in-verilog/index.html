I2C 协议主写功能的 Verilog 实现|I2C Master Write Function In Verilog

基于 Xilinx Spartan FPGA

Posted by HostenWang on Friday, May 22, 2015

1 概述|Abstract

I2C 是我厂常用的一种通信总线, 本文介绍 I2C 主写功能的 verilog 实现.

最近翻出了以前玩 FPGA 时期的代码, 整理到博客.

说起来还挺怀念以前研究通信协议, 看时序图的时光.

2 环境信息|Environment

Windows 7

3 介绍|Introduction

//==============================================================================
// File Name   : 
// Module Name : 
// Description : made by HostenWang to complete 
// Author      : HostenWang
// Date        : 2011/02/11
// Rev.        : 1.3
// Rev. Date       Description
// 2011-02-11 made by HostenWang to complete I2C master write function
//==============================================================================
module i2c_read(

    // from global 
    clk_sys_100M_i,
    rst_sys_n_i,

    // from data interface
    inf_busy_o, 
    inf_dat_ov,
    inf_vld_o,    
    inf_start_i,// po-sedge to send data 
    inf_adr_iv,

    // from i2c wires
    scl_o,
    // 在子模块实现iout，将iout分为i,o,en三个信号，在顶层将其合并
    sda_i,
    sda_o,
    sda_en// sda_en == 1,sda is output 
);

//==============================================================================
//  Input and output declaration
//==============================================================================

// from global 
input           clk_sys_100M_i;
input           rst_sys_n_i;


// from data interface
output          inf_busy_o; 
output  [11:0]  inf_dat_ov;
output          inf_vld_o;
input           inf_start_i;// po-sedge begin send data 
input   [6:0]   inf_adr_iv;

// from i2c wires
output          scl_o;

input           sda_i;
output          sda_o;
output          sda_en;// sda_en = 1,sda is output 

//==============================================================================
//  Wire and reg declaration
//==============================================================================

// input and output
reg             inf_busy_o;
reg  [11:0]     inf_dat_ov;
//reg             inf_vld_o;
reg             scl_o;
reg             sda_o;
reg             sda_en;


reg [8:0]   c_state;
reg [8:0]   n_state;

reg [1:0]   inf_start_reg;  // 寄存器，检测inf_start_reg的上升沿信号    
reg [7:0]   shift_out_reg;  // adr 和 dat 暂存在该寄存器中，随时钟移出到SDA
reg         rec_ack;        // 接收slave返回ACK应答，为1表明收到slave的ACK信号

reg [14:0]  cnt_bit;// 计数器，计数范围是0 - CLK_DIV，控制每个状态机持续时间，进而控制SDA和SCL

//==============================================================================
//  FSM macro definition
//==============================================================================   
parameter   IDLE        = 9'b0_0000_0001;
parameter   START       = 9'b0_0000_0010;// 检测到数据接口的开始信号，sda做出开始时序
parameter   SEND        = 9'b0_0000_0100;
parameter   RECACK      = 9'b0_0000_1000;// 接收到slave的ack响应
parameter   READ        = 9'b0_0001_0000;// 读取slave的返回数据
parameter   SENDACK     = 9'b0_0010_0000;// 给slave发送响应
parameter   READ2       = 9'b0_0100_0000;// 接收第2次8bit数据
parameter   SENDACK2    = 9'b0_1000_0000;
parameter   FINISH      = 9'b1_0000_0000;// 操作结束，状态回到IDLE以后，形成一个完整sda结束时序
        
 
//==============================================================================
//  calculation definition
//==============================================================================

// TMP275 9-bit resolution,once conversation need time 27.5ms,is 1/27.5 KHz = 0.037KHz = 37Hz
// I2C MIN SCL rate is 1KHz, my typical values is 5KHz

`define I2C_FREQ_5KHZ

`ifdef I2C_FREQ_5KHZ
    parameter   CLK_DIV         = (20000 - 1);//5KHz is 100MHz/20000 
`else
    parameter   CLK_DIV         = (5000 - 1);//20KHz is 100MHz/5000 
`endif


//------------------------------------------------------------------------------------------        
// 检测inf_start_reg的上升沿信号
always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                inf_start_reg <= 2'b0;
            end
        else
            begin
                inf_start_reg <= { inf_start_reg[0],inf_start_i};
            end
    end

wire inf_start_posedge = ( 2'b01 == inf_start_reg);



//------------------------------------------------------------------------------------------        
// cnt_bit 计数范围是0 - CLK_DIV，控制每个状态机持续时间，进而控制SDA和SCL
always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                cnt_bit <= 15'd0;
            end
        else if( IDLE == c_state )
            begin
                cnt_bit <= 15'd0;
            end
        else if( cnt_bit == CLK_DIV )
            begin
                cnt_bit <= 15'd0;
            end
        else
            begin
                cnt_bit <=  cnt_bit + 15'd1;
            end

    end    

//wire bit_clk = ( cnt_bit == CLK_DIV );//测试用




// --------------------------------------------------------------
// 输出SDA数据    
// 进入发送数据（SEND）状态后，计数并移出数据
reg [2:0]   cnt_SEND;// counter for SEND state



always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                cnt_SEND <= 3'd0;
            end
        else if( (SEND == c_state)&&(cnt_bit == CLK_DIV) )
            begin
                cnt_SEND <= cnt_SEND + 3'd1;

            end

    end       

// 进入发送数据（SEND）状态的前一个状态（START），就将inf_dat_iv存入shift_out_reg
// 进入发送数据（SEND）状态，按照8bit长度向MSB移位，MSB最后给SDA数据线

always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                shift_out_reg <= 8'd0;
            end
        else if( (SEND == c_state)&&( cnt_bit == CLK_DIV ) )//进入对应状态，开始移出数据
            begin
                shift_out_reg <= ( shift_out_reg << 1 );

            end
        else if( START == c_state  )
            begin
                shift_out_reg <= { inf_adr_iv,1'b1};
            end
    end       


//------------------------------------------------------------------------------------------
// 确认 slave 返回 ACK 信号

always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                rec_ack <= 1'd0;
            end

        else if( (RECACK == c_state) && (1'b0 == sda_i)&&(cnt_bit == CLK_DIV/2)  )
            begin
                rec_ack <= 1'b1;
            end
        else if(RECACK != c_state)
            begin
                rec_ack <= 1'd0;
            end
    end          

//------------------------------------------------------------------------------------------
// 移入SDA线上数据   
reg [2:0]   cnt_READ;// counter for SEND state
reg [7:0]   shift_in_reg;


always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                cnt_READ <= 3'd0;
            end
        else if( (cnt_bit == CLK_DIV)&&( (READ == c_state)||(READ2 == c_state)  ) )
            begin
                cnt_READ <= cnt_READ + 3'd1;

            end
    end       

// 在READ和READ2状态，采样SDA
always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                shift_in_reg <= 8'd0;
            end
        else if( (cnt_bit == CLK_DIV)&&( (READ == c_state) || (READ2 == c_state)  ) )
            begin
                shift_in_reg <= { shift_in_reg[6:0],sda_i};
            end
        else if( IDLE == c_state )
            //在读数据之前将接收数据寄存器清空
            begin
                shift_in_reg <= 8'd0;
            end
    end 


//------------------------------------------------------------------------------------------
// refresh sates


always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                c_state <= IDLE;
            end
        else if( (IDLE != c_state)&&(cnt_bit == CLK_DIV) )// 每个状态机持续时间为1/5KHz的整数倍
            begin
                c_state <= n_state;
            end
        else if(IDLE == c_state)//IDLE状态时，条件满足后马上切换到START状态
            begin
                c_state <= n_state;
            end            
    end    

// switch states

always@(*)
    begin
        case(c_state)
            IDLE:
                begin
                    if(1'b1 == inf_start_posedge) 
                        begin
                            n_state <= START;
                        end
                    else
                        begin
                            n_state <= IDLE;
                        end
                end
            START:   
                begin
                    n_state <= SEND;
                end
            SEND:    
                begin
                    if( 3'b111 == cnt_SEND ) 
                        begin
                            n_state <= RECACK;
                        end
                    else
                        begin
                            n_state <= SEND;
                        end
                end
            RECACK:   
                begin
                    if(1'b1 == rec_ack)
                        begin
                            n_state <= READ;//收到slave的ACK信号，下一状态开始读数据
                        end
                    else
                        begin
                            n_state <= FINISH;//没有收到slave的ACK信号，传输终止
                            //n_state <= READ;//没有收到slave的ACK信号，一直等待
                        end


                end

            READ:     
                begin
                    if( 3'b111 == cnt_READ ) 
                        begin
                            n_state <= SENDACK;
                        end
                    else
                        begin
                            n_state <= READ;
                        end
                end
            SENDACK:  
                begin
                    n_state <= READ2;
                end

            READ2:     
                begin
                    if( 3'b111 == cnt_READ ) 
                        begin
                            n_state <= SENDACK2;
                        end
                    else
                        begin
                            n_state <= READ2;
                        end
                end
            SENDACK2:  
                begin
                    n_state <= FINISH;
                end

            FINISH:   
                begin
                    n_state <= IDLE;
                end
            default:
                begin
                    n_state <= IDLE;
                end


        endcase
    end        


// output sda_o and sda_en
always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                sda_o   <= 1'b1;
                sda_en  <= 1'b1;// default is output and output 1

            end
        else
            begin
                case(c_state)
                    IDLE:
                        begin
                            sda_o       <= 1'b1;
                            sda_en      <= 1'b1;// default is output and output 1
                        end
                    START:    
                        begin
                            sda_o       <= 1'b0;
                            sda_en      <= 1'b1;// default is output and output 1
                        end
                    SEND:     
                        begin
                            sda_o       <= shift_out_reg[7];
                            sda_en      <= 1'b1;// default is output and output 1                               
                        end

                    RECACK:   
                        begin
                            // release sda for read data
                            sda_o       <= 1'bz;
                            sda_en      <= 1'b0;
                        end
                    READ:     
                        begin
                            // release sda for read data
                            sda_o       <= 1'bz;
                            sda_en      <= 1'b0;
                        end

                    SENDACK:   
                        begin

                            sda_o               <= 1'b0;
                            sda_en              <= 1'b1; 
                        end
                    READ2:     
                        begin
                            // release sda for read data
                            sda_o       <= 1'bz;
                            sda_en      <= 1'b0;
                        end

                    SENDACK2:  
                        begin
                            // force sda to 1
                            sda_o               <= 1'b1;
                            sda_en              <= 1'b1;  
                        end
                    FINISH:   
                        begin
                            // force sda to 0
                            sda_o       <= 1'b0;
                            sda_en      <= 1'b1;   

                        end
                    default:
                        begin
                            sda_o   <= 1'b1;
                            sda_en  <= 1'b1;// default is output and output 1

                        end


                endcase
            end        
    end


// output SCL
always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                scl_o <= 1'b1;//默认输出高电平
            end
        else
            begin
                case(c_state)
                    IDLE:
                        begin
                            scl_o <= 1'b1;
                        end
                    START:          //做出SDA下降沿
                        begin
                            if(CLK_DIV/2 == cnt_bit) 
                                begin
                                    scl_o <= 1'b0;
                                end

                        end
                    SEND,           //做出SDA时序
                    RECACK,
                    READ,                                
                    SENDACK,
                    READ2,
                    SENDACK2:  
                        begin
                            if(CLK_DIV/4 == cnt_bit) 
                                begin
                                    scl_o <= 1'b1;
                                end  
                            else if(CLK_DIV*3/4 == cnt_bit) 
                                begin
                                    scl_o <= 1'b0;
                                end  

                        end
                    FINISH:         //做出SDA上升沿
                        begin
                            if(CLK_DIV/4 == cnt_bit) 
                                begin
                                    scl_o <= 1'b1;
                                end  
                        end
                    default:
                        begin
                            scl_o <= 1'b1;
                        end


                endcase
            end        
    end


// output inf_busy_o
always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                inf_busy_o  <= 1'b0;
            end
        else if( START == c_state)
            begin
                inf_busy_o  <= 1'b1;
            end
            
        else if( IDLE == c_state)
            begin
                inf_busy_o  <= 1'b0;
            end
    end        

// output inf_vld_o
assign inf_vld_o = ( (SENDACK2 == c_state)&&(CLK_DIV/2 == cnt_bit) );//keep stable one clock period


//always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
//    begin
//        if(1'b0 == rst_sys_n_i)
//            begin
//                inf_vld_o  <= 1'b0;
//            end
//        else if( SENDACK2 == c_state)
//            begin
//                inf_vld_o  <= 1'b1;
//            end
//            
//        else if( FINISH == c_state)
//            begin
//                inf_vld_o  <= 1'b0;
//            end
//    end        

// output inf_dat_ov
always@(posedge clk_sys_100M_i or negedge rst_sys_n_i)
    begin
        if(1'b0 == rst_sys_n_i)
            begin
                inf_dat_ov  <= 12'd0;
            end
        else if( SENDACK == c_state)
            begin
                inf_dat_ov[11:4]    <= shift_in_reg;
            end
        else if( SENDACK2 == c_state)
            begin
                inf_dat_ov[3:0]     <= shift_in_reg[7:4];
            end                       
        else if( FINISH == c_state)
            begin
                inf_dat_ov  <= 12'd0;
            end
            
    end        



endmodule
