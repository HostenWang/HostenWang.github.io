VBA的强大功能和灵活性使得它成为Office用户和开发人员的重要工具之一。

Sub AddImgWatermark()
'
' Auto Add Watermark to Word
'
    ' 图片路径需要替换为你实际的图片路径，确保图片文件存在并可访问
    imagePath = "D:\Work\Watermark.png"
    
    ' Get page count
    Dim pages As Long
    pages = ActiveDocument.ComputeStatistics(wdStatisticPages)
    Debug.Print "{'page Count':" & pages & "}"
    
    ' Get page height and width
    Set pageSet = ActiveDocument.PageSetup
    Debug.Print "The page height:" & pageSet.PageHeight & ",width:" & pageSet.PageWidth
    
    ' Get page offset
    Dim topOffset As Double
    topOffset = (pageSet.PageHeight - pageSet.PageWidth) / 2
    
    ' focus at top
    For i = 1 To pages Step 1
        Application.Browser.Previous
    Next
    
    ' add image on every page
    For i = 1 To pages Step 1
        If i > 1 Then
            Application.Browser.Next

            ActiveDocument.Shapes.AddShape(MsoAutoShapeType.msoShapeRectangle, 0, topOffset, pageSet.PageWidth, pageSet.PageWidth).Select
            
            Selection.ShapeRange.Name = "WATERMARK_AT_PAGE_" & i
            
            Selection.ShapeRange.Fill.Transparency = 0.85
            Selection.ShapeRange.Fill.Visible = msoTrue
            Selection.ShapeRange.Fill.ForeColor.RGB = RGB(255, 255, 255)
            Selection.ShapeRange.Fill.BackColor.RGB = RGB(255, 255, 255)
            Selection.ShapeRange.Fill.UserPicture imagepath
            
            Selection.ShapeRange.Line.Visible = msoFalse
            
            Selection.ShapeRange.ZOrder msoSendBehindText
        End If

    Next
    
    ' save word file
    If ActiveDocument.Saved = False Then ActiveDocument.Save
End Sub

修改完代码后，保存并关闭VBA编辑器。

在Word文档中，按下Alt + F8键，打开“宏”对话框。

选择刚刚创建的“AddImgWatermark”宏，并点击“运行”按钮。

这样，你的Word文档就会添加一个图片水印。

注意

代码中设置的图片路径需要替换为你实际的图片路径，确保图片文件存在并可访问。

讨论|Discussion

如果要用一张图像的平铺形式来填充形状，请使用 UserTextured 方法。
